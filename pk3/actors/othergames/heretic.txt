ACTOR HereticPsiAmp : CustomInventory replaces ArtiInvisibility
{
	Inventory.PickupMessage "psionic amplifier (1)"
	Inventory.PickupSound "psiamp/pickup"
	Scale 0.3
	States
	{
	Spawn:
		PSIP A -1
		Loop
	Pickup:
		PSIP A 0
		PSIP A 0 A_GiveInventory("PsiAmp",1)
		Stop
	}
}
